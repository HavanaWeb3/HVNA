// Shopify integration example const widget = new HVNAPurchaseWidget('shopify-hvna-widget', { theme: 'light', showHeader: true, minPurchase: 1000 }); // Listen for purchases and add to cart document.getElementById('shopify-hvna-widget').addEventListener('hvna:purchase', (event) => { const { tokens, usdCost } = event.detail; // Add HVNA tokens as Shopify line item fetch('/cart/add.js', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ items: [{ id: 'hvna-token-variant-id', quantity: tokens, properties: { 'Blockchain': 'Base Network', 'Transaction': event.detail.txHash || 'Demo' } }] }) }).then(() => window.location.href = '/cart'); });

React/Vue Integration

// React component example import React, { useEffect, useRef } from 'react'; const HVNAWidget = ({ theme = 'light', onPurchase }) => { const widgetRef = useRef(); const containerRef = useRef(); useEffect(() => { if (window.HVNAPurchaseWidget) { widgetRef.current = new HVNAPurchaseWidget(containerRef.current.id, { theme, showHeader: false, showFooter: false }); containerRef.current.addEventListener('hvna:purchase', onPurchase); } return () => { containerRef.current?.removeEventListener('hvna:purchase', onPurchase); }; }, [theme, onPurchase]); return <div id={`hvna-${Date.now()}`} ref={containerRef} />; };
